    // Updated updateInventory function that sorts items by chance.
    function updateInventory() {
      caseItemsList.innerHTML = '';

      // Convert inventory object to an array of objects with name and count.
      let itemsArray = Object.entries(inventory)
        .filter(([itemName, count]) => count > 0)
        .map(([itemName, count]) => {
          // Find the item in either pool.
          let item = rollItems.find(x => x.name === itemName);
          let isHaunted = false;
          if (!item) {
            item = hauntedLootItems.find(x => x.name === itemName);
            isHaunted = true;
          }
          return { ...item, count, isHaunted };
        });

      // Sort items by chance (ascending)
      itemsArray.sort((a, b) => a.chance - b.chance);

      // Create and append list items for each inventory item.
      itemsArray.forEach(item => {
        const cssClass = item.isHaunted ? 'haunted' : item.rarity.toLowerCase();
        const li = document.createElement('li');
        li.classList.add('stat-item', cssClass);
        li.innerHTML = `
          ${item.name} x${item.count} <span>(Chance: ${item.chance}%)</span>
          <div class="luck-multiplier">Luck: ${item.luckMultiplier}x</div>
          <div class="item-description">${item.description || "No description available."}</div>`;
        caseItemsList.appendChild(li);
      });
    }
    
    // Fixed performRoll function: calculates the total chance of the chosen pool,
    // generates a random number, and then picks an item.
    function performRoll(isHaunted) {
      let pool = isHaunted ? hauntedLootItems : rollItems;
      let totalChance = pool.reduce((sum, item) => sum + item.chance, 0);
      let roll = Math.random() * totalChance;
      let cumulativeChance = 0;
      let selectedItem = 'Nothing!';
      let luckAdded = 0;
      let itemChance = 0;
      // Start with the actual rarity from the item. For haunted rolls, we override it below.
      let itemRarity = '';
      
      for (const item of pool) {
        cumulativeChance += item.chance;
        if (roll < cumulativeChance) {
          selectedItem = item.name;
          luckAdded = item.luckMultiplier || 0;
          itemChance = Math.round(item.chance * 10) / 10;
          itemRarity = item.rarity;
          // Add to inventory using the item's original name
          inventory[item.name] = (inventory[item.name] || 0) + 1;
          totalLuck += luckAdded;
          break;
        }
      }
      
      // Override rarity for haunted rolls to "Haunted"
      if (isHaunted) {
        itemRarity = 'Haunted';
      }
      
      const spookyNote = isHaunted ? " - The spirits have infused your loot..." : "";
      rollResult.textContent = `You rolled: ${selectedItem}${spookyNote}`;
      chanceAndLuck.textContent = `Chance: ${itemChance}% | Luck Added: ${luckAdded}x`;
      totalLuck = Math.round(totalLuck * 10) / 10;
      totalRolls++;
      totalLuckDisplay.textContent = `Total Luck: ${totalLuck}`;
      totalRollsDisplay.textContent = `Total Rolls: ${totalRolls}`;
      updateInventory();
      
      // Apply glow effects. Haunted rolls always use the haunted glow.
      if (isHaunted) {
        rollingArea.classList.add('glow-haunted');
      } else {
        rollingArea.classList.add(`glow-${itemRarity.toLowerCase()}`);
      }
      
      // Apply scaling effect for rare results in standard rolls.
      if (!isHaunted && ['Legendary', 'Mythical', 'Secret'].includes(itemRarity)) {
        rollResult.classList.add('rare-effect');
        setTimeout(() => {
          rollResult.classList.remove('rare-effect');
        }, 1000);
      }
    }
